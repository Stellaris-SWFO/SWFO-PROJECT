firstorder_ship_resurgent = {
	formation_priority = 41
	max_speed = @speed_default
	acceleration = 0.2
	rotation_speed = 0.15
	collision_radius = @titan_collision_radius
	modifier = {
		ship_armor_add = 60000
		ship_shield_add = 60000
	}
	entity = "swfo_resurgent_frame_entity"
	max_hitpoints = 60000
	size_multiplier = 12
	fleet_slot_size = 18
	section_slots = { "bow" = { locator = "part1" } "mid1" = { locator = "part2" } "mid2" = { locator = "part3" } "stern" = { locator = "part4" } }
	num_target_locators = 6
	is_space_station = no
	icon_frame = 5
	base_buildtime = 1000
	can_have_federation_design = no
	enable_default_design = yes	#if yes, countries will have an auto-generated design at start
	
	prerequisites = { tech_RESURGENTCLASS }
	
	combat_disengage_chance = 1.25
	class = shipclass_military
	construction_type = starbase_shipyard
	
	#Komponenten_rechts
	required_component_set = "power_core"
	required_component_set = "ftl_components"
	required_component_set = "thruster_components"
	required_component_set = "sensor_components"
	required_component_set = "combat_computers"		

	potential_construction = {
		OR = {
			AND = {
				is_scope_type = starbase
			}
			AND = {
				is_scope_type = megastructure
				OR = {
					is_megastructure_type = mega_shipyard_3
					is_megastructure_type = mega_shipyard_restored
				}
			}
		}
	}

	possible_construction = {
		custom_tooltip = {
			fail_text = starbase_citadel_trigger
			OR = {
				AND = {
					is_scope_type = megastructure
					OR = {
						is_megastructure_type = mega_shipyard_3
						is_megastructure_type = mega_shipyard_restored
					}
				}
				AND = {
					is_scope_type = starbase
					has_starbase_size >= starbase_citadel
				}
			}
		}
		custom_tooltip = {
			fail_text = need_stardestroyershipyard
			OR = {
				AND = {
					is_scope_type = megastructure
					OR = {
						is_megastructure_type = mega_shipyard_3
						is_megastructure_type = mega_shipyard_restored
					}
				}
				AND = {
					is_scope_type = starbase
					has_starbase_building = stardestroyer_yards
				}
			}
		}
	}
	
	resources = {
		category = ships
		upkeep = {
			energy = 2
			alloys = 2
		}
	}

	min_upgrade_cost = {
		alloys = 10
	}
}
